import { withGitHubAlert, Callout, Blockquote } from 'nextra/components'

const AlertBlockquote = withGitHubAlert(
  ({ type, children }) => <Callout type={type}>{children}</Callout>,
  Blockquote
)

export function useMDXComponents(components) {
  return {
    ...components,
    blockquote: AlertBlockquote
  }
}

# Welcome to JellyDocs!

### JellyDocs is a guide to FTC Software by Team 4654 Jellyfish.

JellyDocs' purpose is to improve team sustainability by centralizing programming knowledge in easy to understand documentation. It covers various aspects of programming, from API reference tables to useful code snippets.

> [!NOTE]
> JellyDocs is always a work in progress, and experienced programmers are encouraged to add or improve pages. However, be sure to check in with the Software Officer before making large changes.

> [!IMPORTANT]
> The goal of the JellyDocs is currently to be kept private within the team.
